1/**
2 * @file mainwindow.h
3 * @brief Ventana principal del HMI para la cinta clasificadora.
4 *
5 * @details Interfaz grafica de usuario (GUI) construida con Qt que permite:
6 * - Conexion/desconexion al puerto serie del ATmega328P.
7 * - Modo Calibracion: medir tiempos de arribo S0->S1/S2/S3,
8 * test manual de servos, ajuste de angulos y umbrales de distancia.
9 * - Modo Clasificacion: iniciar/detener la cinta, configurar
10 * tipos de caja por salida, visualizar estado de sensores IR,
11 * distancia medida, servos activos y conteo de cajas clasificadas.
12 * - Log de comunicacion en tiempo real con formato timestamp.
13 *
14 * La interfaz se divide en:
15 * - Toolbar superior: conexion serie y estado.
16 * - Sidebar izquierdo: navegacion entre paginas.
17 * - Area central: paginas de Calibracion y Clasificacion.
18 * - Panel inferior: log de comunicacion (siempre visible).
19 *
20 * @author Luciano
21 * @date Mayo 2026
22 */
23

41/**
42 * @brief Representa una caja en transito sobre la cinta transportadora.
43 *
44 * @details Se crea al recibir EVT_NEW_BOX del MCU y se elimina al recibir
45 * EVT_SERVO_FIRED para el servo correspondiente.
46 */
47struct BoxInTransit {
48 int id; /**< Identificador secuencial unico de la caja. */
49 uint8_t type; /**< Tipo de caja (altura en cm: 6, 8 o 10). */
50 uint8_t servo; /**< Servo destino asignado (0, 1 o 2). */
51 double totalTime; /**< Tiempo total de arribo en segundos. */
52 double remaining; /**< Tiempo restante estimado en segundos. */
53};
