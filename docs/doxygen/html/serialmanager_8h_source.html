1/**
2 * @file serialmanager.h
3 * @brief Modulo de comunicacion serie con protocolo UNER para el HMI Qt.
4 *
5 * @details Encapsula la comunicacion con el ATmega328P a traves de un puerto
6 * serie (115200 baud, 8N1). Implementa:
7 * - Envio de tramas UNER con checksum XOR.
8 * - Recepcion y decodificacion mediante FSM identica a la del MCU.
9 * - Heartbeat periodico (CMD_ALIVE) cada 500 ms.
10 *
11 * Formato de trama UNER:
12 * @code
13 * [ 'U' | 'N' | 'E' | 'R' | LEN | ':' | CMDID | DATA... | CKS ]
14 * @endcode
15 * donde LEN = 1 (CMDID) + N (datos) + 1 (CKS), y
16 * CKS = XOR de todos los bytes desde 'U' hasta el ultimo dato.
17 *
18 * @author Luciano
19 * @date Mayo 2026
20 */
21
