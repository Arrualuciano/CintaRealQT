57/**
58 * @brief Abre el puerto serie con la configuracion requerida por el MCU.
59 *
60 * @details Configuracion: 115200 baud, 8 bits de datos, sin paridad, 1 stop bit.
61 * Al conectar exitosamente, reinicia la FSM de recepcion e inicia
62 * el timer de heartbeat (500 ms).
63 *
64 * @param portName Nombre del puerto (ej: "COM3").
65 * @return true si la conexion fue exitosa, false si fallo.
66 */
67bool SerialManager::connectPort(const QString &portName) {
68 if (port->isOpen()) port->close();
69
70 port->setPortName(portName);
71 port->setBaudRate(115200);
72 port->setDataBits(QSerialPort::Data8);
73 port->setParity(QSerialPort::NoParity);
74 port->setStopBits(QSerialPort::OneStop);
75
76 if (!port->open(QIODevice::ReadWrite)) {
77 emit connectionError(port->errorString());
78 return false;
79 }
80
81 fsmState = WAIT_U;
82 aliveTimer->start(500);
83 emit connected();
84 return true;
85}

112/**
113 * @brief Construye y envia una trama UNER completa al MCU.
114 *
115 * @details Formato de trama:
116 * @code
117 * [ 'U' | 'N' | 'E' | 'R' | LEN | ':' | CMDID | DATA... | CKS ]
118 * @endcode
119 * LEN = 1 (CMDID) + data.size() + 1 (CKS).
120 * CKS = XOR de todos los bytes desde 'U' hasta el ultimo dato.
121 *
122 * @param cmdId Identificador del comando a enviar.
123 * @param data Bytes de datos adicionales (puede estar vacio).
124 */
125void SerialManager::sendCommand(uint8_t cmdId, const QByteArray &data) {
126 if (!port->isOpen()) return;
127
128 uint8_t totalLen = 1 + data.size() + 1; /* CMDID + data + CKS */
129
130 /* Calcular checksum XOR */
131 uint8_t cks = 'U' ^ 'N' ^ 'E' ^ 'R' ^ totalLen ^ ':' ^ cmdId;
132 for (auto b : data) cks ^= static_cast<uint8_t>(b);
133
134 /* Construir trama completa */
135 QByteArray frame;
136 frame.append('U');
137 frame.append('N');
138 frame.append('E');
139 frame.append('R');
140 frame.append(static_cast<char>(totalLen));
141 frame.append(':');
142 frame.append(static_cast<char>(cmdId));
143 frame.append(data);
144 frame.append(static_cast<char>(cks));
145
146 port->write(frame);
147}

165/**
166 * @brief Procesa un byte recibido a traves de la FSM de decodificacion UNER.
167 *
168 * @details La FSM sincroniza con el encabezado "UNER", captura LEN, el
169 * separador ':', y finalmente el payload byte a byte hasta verificar
170 * el checksum. Al completar una trama valida, emite frameReceived().
171 *
172 * @param byte Byte recibido del puerto serie.
173 */
174void SerialManager::processRxByte(uint8_t byte) {
175 switch (fsmState) {
176 case WAIT_U:
177 if (byte == 'U') fsmState = WAIT_N;
178 break;
179 case WAIT_N:
180 fsmState = (byte == 'N') ? WAIT_E : WAIT_U;
181 break;
182 case WAIT_E:
183 fsmState = (byte == 'E') ? WAIT_R : WAIT_U;
184 break;
185 case WAIT_R:
186 fsmState = (byte == 'R') ? WAIT_LEN : WAIT_U;
187 break;
188 case WAIT_LEN:
189 fsmLen = byte;
190 fsmCks = 'U' ^ 'N' ^ 'E' ^ 'R' ^ byte ^ ':';
191 fsmPayload.clear();
192 fsmState = WAIT_COLON;
193 break;
194 case WAIT_COLON:
195 fsmState = (byte == ':') ? RECV_PAYLOAD : WAIT_U;
196 break;
197 case RECV_PAYLOAD:
198 if (fsmLen > 1) {
199 fsmPayload.append(static_cast<char>(byte));
200 fsmCks ^= byte;
201 fsmLen--;
202 } else {
203 /* Byte de checksum: validar e invocar senal */
204 if (fsmCks == byte && fsmPayload.size() >= 1) {
205 uint8_t cmdId = static_cast<uint8_t>(fsmPayload.at(0));
206 QByteArray data = fsmPayload.mid(1);
207 emit frameReceived(cmdId, data);
208 }
209 fsmState = WAIT_U;
210 }
211 break;
212 }
213}
